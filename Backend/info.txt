Database: webservice

Setup query:

CREATE TABLE `webservice`.`data` 
( `id` INT NOT NULL AUTO_INCREMENT , `latitudine` FLOAT NOT NULL ,
`longitudine` FLOAT NOT NULL , `nomePoi` VARCHAR(30) NOT NULL , 
`tipoPoi` ENUM('SPIAGGIA','MONUMENTO','MUSEO','STADIO','BELVEDERE','ALTRO') NOT NULL ,
`ingresso` ENUM('ACCESSOGRATUITO','APAGAMENTO') NOT NULL ,
`valutazione` ENUM('MAIPIU','PASSABILE', 'DISCRETO', 'BELLO', 'INDIMENTICABILE') NOT NULL ,
ENGINE = InnoDB;
